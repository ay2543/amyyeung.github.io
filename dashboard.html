Scatterplot of snowfall against maximum temperature for days with reported snowfall

ny_noaa %>% 
  mutate(
month = factor(month, levels = month.name, ordered = TRUE)
) %>% 
  filter(snow !=0) %>% 
  plot_ly(
    x = ~tmax,
    y = ~snow,
    type = "scatter", mode = "markers",
    alpha = 0.4,
    color = ~month
  )
